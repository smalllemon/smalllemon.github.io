1、面向对象程序设计

概念：这是一种常见的程序设计模式，将需求抽取成一种对象，通过为对象添加属性和方法进而实现应用需求。

2、理解对象

(1)、创建一个对象

    var person = new Object();

    person.name = 'Nicholas';

    person.age = 29;

    person.job = "Software Engineer";

    person.sayHi = function(){
          alert("GOOD MORNING！"+this.name)
    };

(2)、创建一个对象

    var  person = {
      name:"Nicholas",
      age ： 29,
      job ： "Software Engineer",
      sayHi ： function(){
          alert("GOOD MORNING！"+this.name)
      };
    };

(3)、操作一个对象：增、删、改、读 

    a、修改:Object.defineProperty("对象本身","属性的名字","描述性对象")

        如：  var  person = {};

              object.defineProperty(person,"name",{
                     writable:false, //名字的属性值是可读的
                     value:"Nicholas"
              })

              alert(person.name);  //Nicholas
              person.name = 'Greg';
              alert(person.name);  //Nicholas

            //定义对象的某些属性时，configurable,enumerable,writable三个参数可供选择。


    b、读与写:getter setter 读和设置对象的属性值


        var  book = {
           _year:2004,
           edition:1
        };

        Object.defineProperty(book,"year",{
            get:function(){
                return this._year;
            },
            set:function(newValue){
                if(newValue > 2004){
                    this._year = newValue;
                    this.edition += newValue - 2004;
                }
            };
        })

        book.year = 2005;
        alert(book.edition); //2

    c、定义多个属性

      var  book = {} ;
      object.defineProperties(book,{
          _year:{
              value:2004
          },
          edition:{
            value:1
          },
          year:{
            get:function(){
               return this._year;
            },
            set:function(newValue){
                if(newValue > 2004){
                    this._year  = newValue;
                    this.edition += newValue-2004;
                },
            },
          }
      })

3、创建一个对象

a、工厂模式

  function createPerson(name,age,job){
        var o = new Object();
        o.name = name;
        o.age = age;
        o.job = job ;
        o.sayName =  function(){
           alert(this.name);
        };

        return  0;
  }

  var person = createPerson("Alice",29,"software engineer");


b、构造函数模式

  function Person (name,age,job){
      this.name = name;
      this.age = age;
      this.job = job;
      this.sayName = function(){
        alert(this.name);
      };
  } 

  var person_one = Person("Alice1",29,"software engineer");
  var person_two = Person("Alice2",29,"software engineer");


  //构造函数(具有contructor属性)

  console.log(person_one.contructor == Person ); //true
  console.log(person_two.contructor == Person ); //true

  //构造函数(具有instanceof属性)

  console.log(person_one.instanceof == Person ); //true
  console.log(person_two.instanceof == Person ); //true 

c、原型模式(prototype属性)

  function Person(){

 }
  Person.prototype.name = "Nicholas";
  Person.prototype.age = 29;
  Person.prototype.jod = "teacher";
  Person.prototype.sayName = function(){
     alert(this.name);
  };

  var person1 = new Person();
  person1.sayName();  //Nicholas

  var person2 = new Person();
  person2.sayName();  //Nicholas

  alert(person1.sayName == person2.sayName);  //true

  //测试方法字段属性：  isPrototypeOf()  
  //获取对象的属性字段：getPrototypeOf()
  //检测对象是否含有自我定义的属性时：hasOwnProperty()

d、两种结合着使用

  function Person(name,age,job){
     this.name = name;
     this.age = age;
     this.job = job;
     this.friends = ['Shelby','Court'];
  }

  Person.prototype = {
    contructor: Person,
    SayName:function(){
       alert(this.name);
    },
  }

  var Person1 = new Person("Nicholas",29,"teacher");
  var Person2 = new Person("Greg",29,"Doctor");

  //以上这种方式是将两种开发模式进行了混合


e、动态原型的方法

  function Person(name,age,job){
    this.name = name ;
    this.age = age;
    this.job = job;
    if(typeof  this.sayName != "function"){
       Person.prototype.sayName = function(){
           alert(this.name);
       }

  }
