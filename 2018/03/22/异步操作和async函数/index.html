异步编程

1、异步

概念：一个任务被分成两段执行。

实现方式：

        1、回调函数

        2、事件监听

        3、发布/订阅

        4、promise对象

2、回调函数

概念：保存任务，便于之后的执行。

代码：
fs.readFile(“/ect/passwd”,function(err,data){
if(err){
throw err;
}
console.log(data);
});

3、Generator函数

作用：暂停的地方都用yield,交出函数执行权。

代码：

function gen(x){
   var y = yield x+2 ;
   return y;
}

var g = gen(1);
g.next(); //{value:3 ,done : false}
g.next(); //{value:undefined,done:true}

分析：使用Generator函数会返回一个指针g.并调用next方法会返回一个对象，
用来表示当前阶段的信息。
value返回表达式的值,done表示是否执行完毕。

用法：

实例一:

function * gen(x){
  try{
    var y = yield x+2;
  }catch(e){
    console.log(e);
  }
  return y;
}

//在该函数中也可以进行错误代码的处理。


实例二：

var fetch = require("node-fetch");
function * gen(){
    var url = "http://",
    var result = yield fetch(url);
    console.log(result.bio);
}

var g =  gen();
var result = g.next();

result.value.then(function(data){
    return data.json();
}).then(function(data){
    g.next(data);
});

//调用了一个远程的API，然后从json格式中去获取数据信息。

4、Thunk函数

作用:解决何时进行参数的处理，保存参数整体替换的中间函数。

用法：

function g(m){
    return  m * 2;
}

f(x+4);

//此处如果应用thunk函数

var  thunk = function(){
  return x+4;
}

function f(thunk){
  return thunk()*2;
}

5、Generator函数与Thunk函数结合使用

作用:实现Generator函数自动流程管理。

代码实例：

var fs = require("fs");
var thunkify = require("thunkify");
var readFile = thunkify(fs.readFile);

var gen = function *(){
  var f1 = yield  readFile("filePath");
  console.log(f1.toString());
  var f2 = yield  readFile("filePath");
  console.log(f2.toString());
}

var g = gen();
var r1 = g.next();
r1.value.then(function(err,data){
    if(err){
          throw err;
    }
    var r2 =g.next(data);
    r2.value.then(function(err,data){
      if(err){
          throw err;
      } 
      g.next(data);
    });
});

//发现上述的代码都是基于value反复传入，借助Thunk函数实现自动流程管理。

function run(fn){

      //保存函数的参数
      var gen = fn;
      function next(err,data){
          var  result = gen.next(data);
          if(result.done){
              return;
          }
          result.value(next);
      }
      next();
}

run(gen);

6、co模块

作用:自动执行Generator函数。

实例：

var co = require("co");
co(gen); //返回一个promise对象
co(gen).then(function(){
   console.log("函数执行完成!");
});

co源码: 

function co(gen){
    var ctx = this ;
    return new Promise(function(resolve,reject){
      if(tyoeof gen == "function"){
            gen = gen.call(ctx);
      }
      if(!gen || typeof gen.next !=="function"){
            return  resolve(gen);
      }

      //封装Generator函数中的next方法
      onFulfilled();
      function  onFulfilled(res){
          var ret;
          try{
            ret = gen.next(res);
          }catch(e){
            return reject(e);
          }
          next(ret);
      }

      //反复调用自身的next方法

      function next(ret){
        if(ret.done){
            return resolve(ret.value);
        }
        var value = toPromise.call(ctx,ret.value);
        if(value && isPromise(value)){
            return value.then(onFulfilled,onRejected);                    
        }else{
           return onRejected(new TypeError)
        }
      }

    });
}
