1、原型链

(1)、核心思想

   利用原型让一个引用类型继承另一个引用类型的属性方法。

   原型链的由来：让原型对象等于另一个类型的实例？

   原型对象将包含一个指向另一个原型的指针，同理，另一个原型中也会包含着一个指向

   另一个构造函数的指针，如何这种假设成立，那么层层递进的关系就构成了实例与原型的链条。


  function SuperType(){
    this.pertype = true ;
  }

  SuperType.prototype.getSuperValue = function(){
       return this.pertype;
  }

  //重新定义了一个函数
  function SubType(){
    this.subproperty = false;
  }

  //该函数继承了SuperType
  SubType.prototype = new SuperType();

  SubType.prototype.getSubValue = function(){
      return  this.subproperty;
  }

  var instance = new SubType();
  alert(instance.getSubValue());  //true


  //继承SuperType的SubType,实例instance指向SubType,SubType
    的原型又指向SuperType的原型。

2、原型链中存在的问题

(1)、所有的实例都会共享一个属性，自然也会修改这个属性

  function  getColor(){
      this.colors = [ 'red','blue','green'];
  }

  function subType(){

  }

  subType.protype = new getColor();
  var instance1 = new SubType();
  instance1.colors.push("black");
  console.log(instance1.colors); //'red','blue','green','black'


  var instance2 = new SubType(); 
  console.log(instance2.colors); //'red','blue','green','black'


(2)、解决方法：借用构造函数

  function SuperType(){
    this.colors = ['red','blue','green','black']
  }

  function SubType(){
    SuperType.call(this);
  }

  var instance1 = new SubType();
  instance1.colors.push("black");
  console.log(instance1.colors); //'red','blue','green','black'

  var instance2 = new SubType(); 
  console.log(instance2.colors); //'red','blue','green'


  //SubType具有自己的colors属性副本

3、组合继承

function SuperType(name){
this.name = name ;
this.colors = [“red”,”blue”,”green”];
}

SuperType.prototype.sayName = function (){
alert(this.name);
}

function SubType(name,age){
SuperType.call(this.name);
this.age = age;
}

SubType.prototype.sayAge = function(){
alert(this.age);
}

var instance1 = new SubType(“hi”,29);
instance1.colors.push(“black”);
console.log(instance1.colors); //“red”,”blue”,”green”,”black”
instance1.sayName(); //hi
instance1.sayAge(); //29

var instance2= new SubType(“hello”,27);
console.log(instance2.colors); //“red”,”blue”,”green”,”black”
instance2.sayName(); //hello
instance2.sayAge(); //27
